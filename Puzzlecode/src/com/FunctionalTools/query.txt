 SELECT CNPO.RSRC_ID, CNPO.RSRC_STARTDATE, CNPO.PKG_ID, CNPO.PKG_STARTDATE, 
 CNPO.NPO_DFU_ID, CNPO.DFU_STARTDATE, CNPO.USAGE, CNPO.AUX_USAGE 
 FROM     PROCESSCFE CNPO , NPO_RESRC RSRC 
 WHERE    CNPO.RSRC_ID = ? AND CNPO.RSRC_ID = RSRC.RESRC_ID 
 ORDER BY CNPO.RSRC_ID, CNPO.RSRC_STARTDATE, CNPO.PKG_ID, CNPO.PKG_STARTDATE,CNPO.NPO_DFU_ID, CNPO.DFU_STARTDATE
