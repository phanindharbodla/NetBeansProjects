SELECT DISTINCT RSRCDT_INTERVAL.RESRC_ID,RSRCDT_INTERVAL.STARTDATE,RSRCDT_INTERVAL.POST_DATE,RSRCDT_INTERVAL.POST_DATE_DUR 
 FROM  PROCESSNPO PNPO,NPO_RESRC_DT_INTERVAL RSRCDT_INTERVAL 
 WHERE PNPO.PROCESSID = ? 
 AND PNPO.ACTION_NUMBER = ? 
 AND RSRCDT_INTERVAL.POST_DATE >= ?
 AND PNPO.RSRC_ID = RSRCDT_INTERVAL.RESRC_ID 
 AND PNPO.RSRC_STARTDATE = RSRCDT_INTERVAL.STARTDATE 
 ORDER BY RSRCDT_INTERVAL.RESRC_ID,RSRCDT_INTERVAL.STARTDATE,RSRCDT_INTERVAL.POST_DATE,RSRCDT_INTERVAL.POST_DATE_DUR
